// ---------- INDEXEDDB ----------
let db = null;
const DB_NAME = 'LibraryDB';

function openDB() {
  return new Promise((resolve) => {
    const req = indexedDB.open(DB_NAME, 2);
    req.onupgradeneeded = (e) => {
      const dbRef = e.target.result;
      if (!dbRef.objectStoreNames.contains('books')) {
        dbRef.createObjectStore('books', { keyPath: 'id', autoIncrement: true });
      }
      if (!dbRef.objectStoreNames.contains('purchases')) {
        dbRef.createObjectStore('purchases', { keyPath: 'id', autoIncrement: true });
      }
    };
    req.onsuccess = () => { db = req.result; resolve(); };
    req.onerror = () => resolve();
  });
}

async function getAllBooks() {
  return new Promise(r => {
    const tx = db.transaction('books', 'readonly');
    tx.objectStore('books').getAll().onsuccess = e => r(e.target.result || []);
  });
}
async function saveBook(book) {
  return new Promise(r => {
    const tx = db.transaction('books', 'readwrite');
    tx.objectStore('books').put(book).onsuccess = () => r();
  });
}
async function deleteBook(id) {
  return new Promise(r => {
    const tx = db.transaction('books', 'readwrite');
    tx.objectStore('books').delete(id).onsuccess = () => r();
  });
}

// ---------- STATE ----------
let books = [];
let purchasedIds = [];
let editingId = null;

function showToast(msg) {
  const t = document.getElementById('toast');
  t.innerText = msg;
  t.style.opacity = '1';
  setTimeout(() => t.style.opacity = '0', 2500);
}

// Render Store
async function renderStore() {
  const grid = document.getElementById('store');
  grid.innerHTML = '';
  if (books.length === 0) {
    grid.innerHTML = '<div style="grid-column:1/-1; text-align:center; padding:60px;">📭 No books yet. Add from Admin.</div>';
    return;
  }
  for (let book of books) {
    const priceText = parseFloat(book.price) === 0 ? 'Free' : 'Rs.' + book.price;
    let coverHtml = '';
    if (book.coverData) {
      coverHtml = `<img src="${book.coverData}">`;
    } else if (book.imageUrl) {
      coverHtml = `<img src="${book.imageUrl}" onerror="this.style.display='none'">`;
    } else {
      coverHtml = `<span>${book.emoji || '📗'}</span>`;
    }
    const card = document.createElement('div');
    card.className = 'book-card';
    card.innerHTML = `
      <div class="cover" style="background:linear-gradient(135deg, ${book.color1||'#1A1F4D'}, ${book.color2||'#5B2EFF'});">
        ${coverHtml}
      </div>
      <h4>${escapeHtml(book.title)}</h4>
      <div class="price">${priceText}</div>
    `;
    card.onclick = () => openBookDetail(book.id);
    grid.appendChild(card);
  }
}

// Open Book Detail with Background Image (Wallpaper) or Gradient
async function openBookDetail(bookId) {
  const book = books.find(b => b.id === bookId);
  if (!book) return;
  const isPurchased = purchasedIds.includes(book.id);
  const isFree = parseFloat(book.price) === 0;
  
  document.getElementById('store').style.display = 'none';
  document.getElementById('detail').style.display = 'block';
  
  // Apply background: if background image exists, use it; else fallback to gradient
  if (book.backgroundData) {
    document.body.style.background = `url("${book.backgroundData}") center center / cover no-repeat fixed`;
  } else {
    document.body.style.background = `linear-gradient(135deg, ${book.color1||'#1A1F4D'}, ${book.color2||'#5B2EFF'})`;
  }
  
  // Cover image
  const coverDiv = document.getElementById('detailCover');
  if (book.coverData) coverDiv.innerHTML = `<img src="${book.coverData}">`;
  else if (book.imageUrl) coverDiv.innerHTML = `<img src="${book.imageUrl}" onerror="this.style.display='none'">`;
  else coverDiv.innerHTML = `<span>${book.emoji || '📗'}</span>`;
  coverDiv.style.background = `linear-gradient(135deg, ${book.color1}, ${book.color2})`;
  
  document.getElementById('detailTitle').innerText = book.title;
  document.getElementById('detailAuthor').innerText = '✍️ ' + (book.author || 'Unknown');
  document.getElementById('detailDesc').innerText = book.desc || '';
  document.getElementById('detailPrice').innerText = isFree ? 'Free' : `Rs. ${book.price}`;
  
  const readBtn = document.getElementById('readBtn');
  if (isFree || isPurchased) {
    readBtn.innerText = '📖 Read Now';
    readBtn.onclick = () => openPDF(book.id);
  } else {
    readBtn.innerText = '💰 Buy Now (Demo)';
    readBtn.onclick = () => {
      if (confirm(`Demo payment for ${book.title}? (No actual charge)`)) {
        purchasedIds.push(book.id);
        showToast('Purchase successful! You can now read.');
        openBookDetail(book.id);
      }
    };
  }
}

// Open PDF
function openPDF(bookId) {
  const book = books.find(b => b.id === bookId);
  if (!book || !book.pdfData) {
    showToast('❌ No PDF uploaded for this book.');
    return;
  }
  try {
    const binary = atob(book.pdfData.split(',')[1]);
    const arr = new Uint8Array(binary.length);
    for (let i = 0; i < binary.length; i++) arr[i] = binary.charCodeAt(i);
    const blob = new Blob([arr], {type: 'application/pdf'});
    const url = URL.createObjectURL(blob);
    const win = window.open();
    if (!win) {
      showToast('⚠️ Popup blocked. Allow popups for this site.');
      return;
    }
    win.document.write(`
      <html><head><title>${escapeHtml(book.title)}</title></head>
      <body style="margin:0;"><iframe src="${url}" width="100%" height="100%" style="border:none;"></iframe></body>
    </html>
    `);
  } catch(e) {
    showToast('PDF error');
  }
}

let adminLoggedIn = false;

    const pass = prompt('Enter admin password:');

      adminLoggedIn = true;
      document.getElementById('adminPanel').style.display = 'block';
      renderAdminList();
    } else {
      alert('Wrong password');

  } else {
    const panel = document.getElementById('adminPanel');
    panel.style.display = panel.style.display === 'block' ? 'none' : 'block';
    if (panel.style.display === 'block') renderAdminList();
  }
}

async function renderAdminList() {
  const listDiv = document.getElementById('adminBookList');
  listDiv.innerHTML = '';
  for (let book of books) {
    const div = document.createElement('div');
    div.style.cssText = 'display:flex; justify-content:space-between; background:rgba(255,255,255,0.05); padding:8px; margin:4px 0; border-radius:40px;';
    div.innerHTML = `
      <span>📘 ${escapeHtml(book.title)}</span>
      <div>
        <button onclick="editBook(${book.id})" style="margin-right:4px;">✏️</button>
        <button onclick="deleteBookHandler(${book.id})">🗑️</button>
      </div>
    `;
    listDiv.appendChild(div);
  }
}

window.editBook = async (id) => {
  const book = books.find(b => b.id === id);
  if (!book) return;
  editingId = id;
  document.getElementById('title').value = book.title;
  document.getElementById('author').value = book.author || '';
  document.getElementById('price').value = book.price;
  document.getElementById('imageUrl').value = book.imageUrl || '';
  document.getElementById('emoji').value = book.emoji || '📗';
  document.getElementById('color1').value = book.color1 || '#1A1F4D';
  document.getElementById('color2').value = book.color2 || '#5B2EFF';
  document.getElementById('description').value = book.desc || '';
  // Clear file inputs (cannot preselect)
  document.getElementById('coverFile').value = '';
  document.getElementById('bgFile').value = '';
  document.getElementById('pdfFile').value = '';
  showToast('Edit mode: change data and click Save');
};

window.deleteBookHandler = async (id) => {
  if (confirm('Delete this book?')) {
    await deleteBook(id);
    await loadBooks();
    renderStore();
    renderAdminList();
    showToast('Book deleted');
  }
};

async function saveBookHandler() {
  const title = document.getElementById('title').value.trim();
  if (!title) { showToast('Title required'); return; }
  
  let coverData = null;
  const coverFile = document.getElementById('coverFile').files[0];
  if (coverFile) {
    coverData = await new Promise(r => { const fr = new FileReader(); fr.onload = e => r(e.target.result); fr.readAsDataURL(coverFile); });
  }
  
  let bgData = null;
  const bgFile = document.getElementById('bgFile').files[0];
  if (bgFile) {
    bgData = await new Promise(r => { const fr = new FileReader(); fr.onload = e => r(e.target.result); fr.readAsDataURL(bgFile); });
  }
  
  let pdfData = null;
  const pdfFile = document.getElementById('pdfFile').files[0];
  if (pdfFile) {
    pdfData = await new Promise(r => { const fr = new FileReader(); fr.onload = e => r(e.target.result); fr.readAsDataURL(pdfFile); });
  }
  
  const book = {
    title: title,
    author: document.getElementById('author').value.trim(),
    price: document.getElementById('price').value || '0',
    imageUrl: document.getElementById('imageUrl').value.trim(),
    coverData: coverData,
    backgroundData: bgData,
    emoji: document.getElementById('emoji').value.trim() || '📗',
    color1: document.getElementById('color1').value,
    color2: document.getElementById('color2').value,
    desc: document.getElementById('description').value.trim(),
    pdfData: pdfData
  };
  
  if (editingId) {
    book.id = editingId;
    const old = books.find(b => b.id === editingId);
    if (!coverData && old && old.coverData) book.coverData = old.coverData;
    if (!bgData && old && old.backgroundData) book.backgroundData = old.backgroundData;
    if (!pdfData && old && old.pdfData) book.pdfData = old.pdfData;
    editingId = null;
  }
  await saveBook(book);
  await loadBooks();
  renderStore();
  renderAdminList();
  showToast(book.id ? 'Book updated' : 'Book added');
  
  // Reset form (keep admin panel open)
  document.getElementById('title').value = '';
  document.getElementById('author').value = '';
  document.getElementById('price').value = '0';
  document.getElementById('imageUrl').value = '';
  document.getElementById('emoji').value = '📗';
  document.getElementById('color1').value = '#1A1F4D';
  document.getElementById('color2').value = '#5B2EFF';
  document.getElementById('description').value = '';
  document.getElementById('coverFile').value = '';
  document.getElementById('bgFile').value = '';
  document.getElementById('pdfFile').value = '';
}

}

async function loadBooks() {

  purchasedIds = []; // in real app, load from DB
}

function escapeHtml(str) {
  if (!str) return '';
  return str.replace(/[&<>]/g, m => ({'&':'&amp;','<':'&lt;','>':'&gt;'}[m]));
}
